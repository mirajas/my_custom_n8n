% header.tex - FINAL POLISHED VERSION

\documentclass[openany]{book} % Using 'openany' to remove blank pages

% --- Load necessary packages ---
\usepackage{geometry}
\usepackage{sectsty}
\usepackage{graphicx}
\usepackage{hyperref}
\usepackage{fancyhdr} % <-- Add this for header/footer control

% --- Set Page Margins ---
\geometry{paper=a4paper, margin=1in}

% --- Style Chapter Headings ---
\chapterfont{\centering\Huge\bfseries}

% --- Configure Page Headers and Footers ---
\pagestyle{fancy} % Apply the fancy style
\fancyhf{} % Clear all existing header and footer fields
\fancyhead[RO,LE]{\thepage} % Put the page number on the outer edge (Right Odd, Left Even)
\renewcommand{\headrulewidth}{0pt} % Remove the line under the header

% --- TOC and Link Styling (keep as is) ---
\let\oldtableofcontents\tableofcontents
\renewcommand{\tableofcontents}{%
  \cleardoublepage
  \oldtableofcontents
  \cleardoublepage
}
\hypersetup{colorlinks=true, linkcolor=blue, urlcolor=cyan}
